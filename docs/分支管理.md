1. 提交代码
    git add .
        # 将所有修改过的文件添加到暂存区，准备提交。
    git commit -m "你的提交说明"
        # 把暂存区的内容提交到本地仓库，并写明本次修改内容。
    git push
        # 将本地仓库的提交推送到远程仓库（如 GitHub），让远程仓库同步最新代码。
            # 可以多次使用 git add . 和 git commit -m "你的提交说明"，最后一次性 git push，把所有本地提交推送到远程仓库。
            # 多次 commit，每次都会生成一个独立的哈希值（commit hash），一次 push 会把所有未推送的 commit（及其哈希）同步到远程仓库。
            # 每个 commit 都可以单独操作和追溯，可用 git log --oneline 查看所有哈希。
            # 如果还没 push，可以回滚到任意一次本地 commit：
            # 1. 用 git log --oneline 查看所有本地提交，找到目标哈希（如 abc123）。
            # 2. 用 git reset --hard abc123 回到该提交，之后的本地 commit 会被删除（不可恢复）。
            # 注意：只要这些 commit 没有 push 到远程，reset --hard 是安全的。
            # 如果只想回退 commit，但保留后续修改内容，可以用：
            # git reset --soft abc123   # 回到指定 commit，后续 commit 的内容会放回暂存区（可重新 commit）
            # git reset --mixed abc123  # 回到指定 commit，后续 commit 的内容会放回工作区（可重新 add 和 commit）

2. 拉取特定版本
    git fetch
        # 从远程仓库获取最新的提交和分支信息，不会自动合并到本地分支，只是同步远程状态。
    git log --oneline -n 20
        # 显示最近 20 条提交记录，每条只显示一行（简短哈希和说明），用于快速浏览提交历史。
    git checkout <commit_hash>
        # 切换到指定的提交版本（<commit_hash>），让工作区回到该历史状态。

            # 注意：如果直接在历史某个版本（分离 HEAD 状态）下开发和提交，新的 commit 没有分支归属，容易丢失或管理混乱。
            # 推荐做法：切换到历史版本后，先新建分支再开发。
            # 示例：
            # git checkout <commit_hash>
            # git checkout -b 新分支名
            # 这样后续开发、提交都在新分支上，安全且易管理。

3. 拉取特定版本后，在这个版本基础上进行后续开发
    git fetch
        # 从远程仓库获取最新的提交和分支信息。
    git log --oneline -n 20
        # 查看最近 20 条提交，找到目标哈希。
    git checkout <commit_hash>
        # 切换到目标版本（分离 HEAD 状态）。
    git checkout -b 新分支名
        # 在当前版本基础上新建并切换到一个分支，后续开发和提交都在这个新分支上进行，安全且易管理。
    git push --set-upstream origin 新分支名
        # 新建分支首次推送时，需用此命令将本地分支推送到远程仓库，并建立跟踪关系。后续只需 git push 即可。
  
    # 查询当前分支和版本
    git branch --show-current
        # 显示当前所在的分支名
    git log --oneline -1
        # 显示当前分支最近一次提交（即当前版本）的简要信息（哈希和说明）

    git show --stat
        # 显示最近一次提交的详细信息，包括完整哈希、作者、时间、提交说明，以及本次提交涉及的文件变动统计。

        # 查询历史分支
        git branch -a
            # 显示所有本地和远程分支（包括历史分支）。
        git branch
            # 只显示本地分支。
        git branch -r
            # 只显示远程分支。
