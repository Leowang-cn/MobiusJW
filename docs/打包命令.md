## 激活虚拟环境
source mobius_env/bin/activate
## 打包命令（适配 macOS 15+，需签名）

pyinstaller --onedir \
            --windowed \
            --name "MobiusJ" \
            --icon=mobiusj/assets/icon.icns \
            --add-data "mobiusj/assets/:assets/" \
            mobiusj/main.py

# -----------------------------
# macOS 15 及以上系统必须对 .app 进行签名，否则无法运行！
# 若仅本机测试，可用 ad-hoc 签名：
codesign --deep --force --sign - dist/MobiusJ.app

# 若有 Apple Developer ID 证书，建议用证书签名：
# codesign --deep --force --sign "Developer ID Application: 你的名字 (XXXXXXXXXX)" dist/MobiusJ.app

# 签名后可用以下命令验证：
codesign --verify --deep --strict --verbose=2 dist/MobiusJ.app

# 注意：签名后不要再修改 .app 内部文件，否则签名会失效。

在不通电脑上各打包一次
- 原版本号为macmini
- a版本号为macbookpro
原因是：macOS 15+ 的严格安全策略。

主要原因
ad-hoc 签名（--sign -）只保证本机可运行，换电脑后会被系统拦截。
macOS 15 及以上系统对未用 Apple Developer ID 证书签名的 .app 直接拒绝运行，报“Code Signature Invalid”。
签名花钱，每年花钱。
